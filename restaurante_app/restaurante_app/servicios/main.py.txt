# main.py

# Importaciones modulares correspondientes
from modelos.producto import Producto
from modelos.cliente import Cliente
from servicios.restaurante import Restaurante

def main():
    print("=== INICIANDO SISTEMA DE GESTIÓN DE RESTAURANTE ===")
    
    # 1. Crear el servicio principal (Restaurante)
    mi_restaurante = Restaurante("Sabores de Python")

    # 2. Crear instancias de Productos y agregarlos al menú
    p1 = Producto("Hamburguesa Artesanal", 8.50)
    p2 = Producto("Papas Fritas con Queso", 4.00)
    p3 = Producto("Limonada Imperial", 2.50)
    p4 = Producto("Pastel de Chocolate", 3.75)

    mi_restaurante.agregar_al_menu(p1)
    mi_restaurante.agregar_al_menu(p2)
    mi_restaurante.agregar_al_menu(p3)
    mi_restaurante.agregar_al_menu(p4)

    # Mostrar menú inicial
    mi_restaurante.mostrar_menu()

    # 3. Crear Clientes y registrarlos en el restaurante
    cliente_juan = Cliente("Juan Pérez")
    cliente_maria = Cliente("María López")

    mi_restaurante.registrar_cliente(cliente_juan)
    mi_restaurante.registrar_cliente(cliente_maria)

    # 4. Simular pedidos (interacción entre objetos)
    print("\n--- Simulando Pedidos ---")
    cliente_juan.ordenar_producto(p1)  # Juan pide hamburguesa
    cliente_juan.ordenar_producto(p3)  # Juan pide limonada
    
    cliente_maria.ordenar_producto(p2) # María pide papas
    cliente_maria.ordenar_producto(p4) # María pide pastel
    cliente_maria.ordenar_producto(p3) # María también pide limonada

    # 5. Mostrar los resultados finales organizados en consola
    mi_restaurante.mostrar_estado_mesas()

if __name__ == "__main__":
    main()
