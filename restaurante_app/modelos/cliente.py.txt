# modelos/cliente.py

class Cliente:
    def __init__(self, nombre: str):
        """
        Constructor de la clase Cliente.
        Cada cliente inicia con un nombre y una lista de consumo vacía.
        """
        self.nombre = nombre
        self.historial_consumo = []

    def ordenar_producto(self, producto):
        """
        Agrega un objeto Producto a la lista de consumo del cliente.
        """
        self.historial_consumo.append(producto)
        print(f"-> {self.nombre} ordenó: {producto.nombre}")

    def calcular_total_cuenta(self):
        """
        Suma los precios de todos los productos ordenados.
        """
        total = sum(producto.precio for producto in self.historial_consumo)
        return total

    def __str__(self):
        """
        Representación en texto del cliente y su estado actual.
        """
        return f"Cliente: {self.nombre} | Productos pedidos: {len(self.historial_consumo)}"
